 66    def RER(self, expr) -> Comment|Complete|Incomplete|Error:
 67        """Read-Evaluate-Return function. Meant to be looped."""
 68        
 69        if PAR.iscommented(expr):
 70            if INT.interpreter.MULTILINE_COMMENT_CLOSE in expr:
 71                if INT.interpreter.COMMENT_COUNTER: INT.interpreter.COMMENT_COUNTER -= 1
 72                else: raise SyntaxError(f"unmatched closing comment in {expr}")
 73            elif INT.interpreter.MULTILINE_COMMENT_OPEN in expr: INT.interpreter.COMMENT_COUNTER += 1
 74
 75            return self.Comment()
 76               
 77        elif PAR.iscomplete(expr):
 78            try:
 79                return self.Complete(self.interpret(expr))
 80            except Exception as e: 
 81                if INT.interpreter.dFlag: raise e                
 82                else: 
 83                    msg = Text(f"{type(e).__name__}: {e}")
 84                    if INT.interpreter.zFlag: msg.append_text(INT.interpreter.del_random_keyword())
 85            
 86                    return self.Error(msg)
 87                                    
 88        return self.Incomplete()
 89

 91    def RERL(self, stream: str = sys.stdin):
 92        """Read-Evaluate-Return-Loop. Wrapper for `RER`. Yields a continuous stream of output (generator)."""
 93
 94        expr = ""
 95
 96        for line in stream:
 97
 98            expr += f"{line}\n"
 99
100            output = self.RER(expr.strip())
101
102            # Reset expression during comments and on completion/error
103            if not isinstance(output, self.Incomplete): expr = ""
104            if not isinstance(output, self.Comment): yield output

107    def REPL(self, stream = sys.stdin, suppress = False) -> None:
108        """Read-Evaluate-Print-Loop. Converts `RERL` output to interactive stdout."""
109
110        # Suppress prompts if reading from file 
111        if suppress: 
112
113            # Completely silent if continuing into interactive shell
114            if INT.interpreter.iFlag: list(self.RERL(stream))
115            # Otherwise print banner
116            else: print(f"--- OPAL v{INT.interpreter.VERSION} ---")
117
118        # If unsuppressed and interactive, proceed as normal
119        elif INT.interpreter.iFlag:
120            print(self.welcome())
121            print(self.prompt(), end='')
122
123        for line in self.RERL(stream):
124            if line.get() != None: print(line.get())
125            if not suppress: print(self.prompt(interim=isinstance(line, self.Incomplete)), end='')

128    def interpret(self, expr: str) -> str | Text:
129        """Fully interpret a complete expression."""
130        
131        # Ignore empty lines
132        if expr == "": return None
133        
134        # Interpret using the Python interpreter
135        elif expr.startswith("python"): return f"{eval(expr.removeprefix("python"))}"
136        
137        # Handle language extensions
138        elif expr.startswith("@start"): return INT.interpreter.extend(expr)
139        
140        # Identify solitary keywords
141        elif KEY.iskeyword(expr): return f"{expr} is an operator, built-in function or reserved word."
142
143        # Match repl-level commands
144        elif expr in self.COMMANDS: return self.COMMANDS[expr]()
145
146        # Otherwise parse the line and convert it to Python syntax, evaluate, and return as an OPAL string 
147        return PAR.toOPAL(EVA.evaluate(PAR.parse(expr)))

296    def show_keywords(self) -> str|Table:
297        """Display all language keywords."""
298
299        if not any(INT.interpreter.KEYWORDS.values()): return "[red]No keywords found."
300
301        padding = 2
302
303        # Calculate the character limit of each word with padding
304        offset = max(max(len(keyword) for keyword in category) for category in INT.interpreter.KEYWORDS.values() if category) + padding
305
306        title = f"KEYWORDS ({INT.interpreter.current_keyword_num()}/{INT.interpreter.INITIAL_KEYWORD_NUM})"
307        msg = ""
308
309        categories = { f"{c} ({len(INT.interpreter.KEYWORDS[c])})" : sorted(INT.interpreter.KEYWORDS[c]) for c in INT.interpreter.KEYWORDS}
310
311        for section_idx, section_title in enumerate(categories):
312
313            # Add spacing between categories
314            if section_idx > 0: msg += "\n\n"
315
316            # Add the section title
317            msg += f"{" "*((offset*3-len(section_title))//2)}{section_title}\n"
318
319            # Iterate through the keywords in the section to organize into columns
320            for keyword_idx, keyword in enumerate(categories[section_title]):
321
322                # Move to the next row every three keywords
323                if keyword_idx % 3 == 0: msg += f"\n{" " * padding}"
324
325                # Add each keyword along with the necessary amount of whitespace to justify the columns
326                msg += f"{keyword}{' ' * (offset-len(keyword))}"
327
328                # If it is the end of a section and the keywords need to be justified
329                column = (keyword_idx + 1) % 3
330                if keyword_idx == len(categories[section_title]) - 1 and column != 0:
331                    # Replace the 'missing words' with whitespace to complete the columns
332                    msg += ' ' * offset * (2 if column == 1 else 1)
333        
334        return Group(
335            "",
336            Panel.fit(
337                msg,
338                title=title,    
339                box=INT.interpreter.DEFAULT_OUTLINE,
340                border_style=INT.interpreter.DEFAULT_COLOR,
341            ),
342            ""
343        )

297    def show_keywords(self) -> str|Table:
298        """Display all language keywords."""
299
300        if not any(INT.interpreter.KEYWORDS.values()): return "[red]No keywords found."
301
302        padding = 2
303
304        # Calculate the character limit of each word with padding
305        offset = max(max(len(keyword) for keyword in category) for category in INT.interpreter.KEYWORDS.values() if category) + padding
306
307        title = f"KEYWORDS ({INT.interpreter.current_keyword_num()}/{INT.interpreter.INITIAL_KEYWORD_NUM})"
308        msg = ""

310        categories = { f"{c} ({len(INT.interpreter.KEYWORDS[c])})" : sorted(INT.interpreter.KEYWORDS[c]) for c in INT.interpreter.KEYWORDS}
311
312        for section_idx, section_title in enumerate(categories):
313
314            # Add spacing between categories
315            if section_idx > 0: msg += "\n\n"
316

318            msg += f"{" "*((offset*3-len(section_title))//2)}{section_title}\n"
